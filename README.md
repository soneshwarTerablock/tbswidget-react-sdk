```bash
npm install --save-dev tbswidget-react-sdk
```

## Usage

```jsx

import { TBSwidget } from "tbswidget-react-sdk";
import 'tbswidget-react-sdk/dist/index.css';

const App = () => {
  return <TBSwidget />
}
```
